        window.players = [];

        function updateUsers() {
            const playerList = document.getElementById("player-list");
            playerList.innerHTML = "";
            window.players.forEach(player => {
                const playerDiv = document.createElement("div");
                playerDiv.innerHTML = `<strong>Nick:</strong> ${player.nick} <br>
                                       <strong>Points:</strong> ${player.points} <br>
                                       <strong>Victories:</strong> ${player.victory} <br>
                                       <strong>Room:</strong> <a href="${player.room}" target="_blank">${player.room}</a> <br>
                                       <strong>Avatar:</strong> ${player.avatar} <br>
                                       <strong>Foto:</strong> ${player.foto} <br>`;
                playerList.appendChild(playerDiv);
            });
        }

            }

        function fetchData() {
            fetch("https://gartic.io/req/list?search=&language[]=" + window.lang)
                .then(response => response.json())
                .then(data => {
                    data.forEach(room => {
                        if (room.quant > 0) {
                            getData(room.code);
                        }
                    });
                });
        }
